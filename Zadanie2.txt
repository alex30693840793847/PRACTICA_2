        private void Button_Click(object sender, RoutedEventArgs e)
        {
            try
            {
                int N = Convert.ToInt32(TBN.Text);
                int g = N % 60;
                OTV.Text = $"Ответ:\n{g}";
            }
            catch (FormatException)
            {
                MessageBox.Show("Введены не корректные данные");
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
        }
